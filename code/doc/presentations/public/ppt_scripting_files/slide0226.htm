Fitting – an example
n   - include 'fitting.g'
n   - x:=seq(100)
n   - y:= 11.3 + 5.6*x + rand.normal(0.0,200.0,100)
n
n   - myfit:=fitter()
n   - myfit.init(n=2)
n   - myfit.makepoly(x,y)
n   - myfit.fit()
n
n   - sol:=myfit.solution();
n   - err:=myfit.error()
n
n   - print sol ; print err
n     [12.4591765 5.60473783]
n     [2.21460569 0.0380726688]
n
n   - yfit:=sol[1] + sol[2]*x
n
n   - mypg:=pgplotter()
n   - mypg.plotxy(x,y,
n   +  plotlines=F,xtitle='X',
n   +  ytitle='Y',title='Y vs. X');
n   - mypg.plotxy(x,yfit,
n   +  plotlines=T,
n   +  newplot=F,linecolor=3)
